8     /* 
9     Items can be added to or removed from the deque at any time, even after leaving an area until the player indicates they have solved the puzzle
10     For the puzzles (see puzzles) the items are dequeued and compared with the order in which they were supposed to be added
11     Uses ArrayDeque over Queue to undo item insertion (from user input) in constant time
12     
13     Items are returned to player inventory and removed from the PuzzleDeque upon failure
14     Items removed from PuzzleDeque and don't go back to player inventory upon success
15     */
16     
17     // Reasons for ArrayDeque over ArrayList:
18     //& • Removal of items for comparisons is simpler and doesn't require keeping track of shifting indices
19     //& • Puzzles poll() items one at a time from submitted PuzzleDeque to avoid iterating through the entire PuzzleDeque and unnecessarily assigning everything to a variable if the puzzle is incorrect before reaching the last item
20     //& • PuzzleDQ pollLast() remaining items directly into the player's inventory with a while loop (takes O(N) based on what's left in the PuzzleDQ)
21     
22     // See Puzzle class for full implementation with user input logic and inventory interaction
23     
24     public class PuzzleDeque {
25         private final ArrayDeque<Item> puzzleDeque;
26     
27         public PuzzleDeque (){
28             this.puzzleDeque = new ArrayDeque<>();
29         }
30     
31         public void add(Item item) {
32             if (item == null) throw new NullPointerException("Cannot add null item to puzzle");
33             if (!(ItemCatalog.masterInventory.contains(item))) throw new IllegalArgumentException("Item must already exist in master item catalog to add to inventories");
34             this.puzzleDeque.add(item);
35         }
36     
37         public Item poll() {
38             return this.puzzleDeque.poll();
39         }
40     
41         public Item undo() {
42             return this.puzzleDeque.pollLast();
43         }
44     
45         public int getSize() {
46             return this.puzzleDeque.size();
47         }
48     
49         public boolean isEmpty() {
50             return this.puzzleDeque.isEmpty();
51         }
52     
53         public boolean contains(Item item) {
54             return this.puzzleDeque.contains(item);
55         }
56     
57         public String toString() {
58             if (this.puzzleDeque.isEmpty()) return "Puzzle is empty.";
59             System.out.println("");
60             StringBuilder sb = new StringBuilder("Current item order: ");
61             for (Item item : puzzleDeque) sb.append(item + "; ");
62             return sb.toString();
63         }
