1     package com.textquest.Inventory_and_Items;
2     import java.util.Collections;
3     import java.util.Map;
4     import java.util.HashMap;
5     import java.lang.StringBuilder;
6     
7     // Inventory is similar to PuzzleDeque in form; almost the same as a HashMap, but with restricted use of HashMap methods as well as custom methods
8     // Uses String keys and Item values for easier user input parsing
9     // GameCharacters, Players, and Rooms are all instantiated with their own inventories
10     
11     public class Inventory{
12         private final HashMap<String, Item> items = new HashMap<>();
13         
14         // Printing full sets takes O(N)
15         
16         public String toString() {
17             StringBuilder items = new StringBuilder("");
18             if (this.items.isEmpty()) return "Currently, there are no items.";
19             for (Item item : this.items.values()) {
20                 if (this.items.values().size() == 1) items.append(item.getName());
21                 else items.append(item.getName() + "; ");
22             }
23             return items.toString();
24         }
25     
26         public boolean isEmpty() {
27             return this.items.isEmpty();
28         }
29     
30         public Map<String, Item> getItems() {
31             return Collections.unmodifiableMap(items);
32         }
33     
34         public void printInventory() {
35             for (Item item : this.items.values()) System.out.println(item);
36         }
37     
38         // Prints list of usable string keys to remind user what items they can use
39         public void printItemNicknames(){
40             System.out.println(this.items.keySet().toString());
41         }
42     
43         public boolean hasItem(String itemNickname) {
44             return this.items.containsKey(itemNickname);
45         }
46     
47         public Item getItem(String itemNickname) {
48             return this.items.get(itemNickname);
49         }
50     
51         // addItem modifies HashMap's put() by only allowing one Item argument. The value assigned is a pre-assigned String "nickname" so that key assignments are controlled and consistent
52         //! Can only add items from ItemCatalog class to any inventory, or it throws an error
53         public void addItem(Item item) {
54             if (item == null) throw new NullPointerException("Cannot add null item to inventory.");
55             if (!(ItemCatalog.masterInventory.contains(item))) throw new IllegalArgumentException("Item must already exist in master item catalog to add to inventories");
56             this.items.put(item.getNickName(), item);
57         }
58     
59         // Search for nickName String key instead of Item value because containsValue() would degrade the method to O(N)
60         // addItem uses Item directly for fast ItemCatalog HashSet lookup conditional - doesn't need String nickName like it would need for HashMap based inventory
61         public void removeItem(String itemNickname) {
62             if (this.items.containsKey(itemNickname)) {
63                 this.items.remove(itemNickname);
64             }
65             else System.out.println(itemNickname + " is not in your inventory!");
66         }
