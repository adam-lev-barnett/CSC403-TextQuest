21         public String pickUpItem(String[] userIn) throws InvalidUserInputException {
22             if (userIn.length == 1) return "Pick what? Your nose? Gross!";
23             else if (userIn.length == 2) {
24                 if (userIn[1].equalsIgnoreCase("up")) return "Pick up what?";
25                 else return "Ya gotta \"pick up\" an item. Picking anything else won't work.";
26             } else if (userIn.length > 2 && userIn[1].equalsIgnoreCase("up")) {
27                 StringBuilder sbItemName = new StringBuilder(userIn[2]);
28                 for (int i = 3; i < userIn.length; i++) {
29                     sbItemName.append(" ").append(userIn[i]);
30                 }
31                 // TODO get the view to return the full inventory
32                 String itemName = sbItemName.toString().toLowerCase();
33                 if (player.getRoom().getRoomInventory().hasItem(itemName)) {
34                     player.addToInventory(player.getRoom().getRoomInventory().getItem(itemName));
35                     player.getRoom().getRoomInventory().removeItem(itemName);
36                     return player.getInventory().getItem(itemName) + " description: " + player.getInventory().getItem(itemName).getDesc();
37                 } else return "There is no " + itemName + " to pick up!";
38             }
39             else throw new InvalidUserInputException();
40         }

45                     StringBuilder sbItemName = new StringBuilder(userIn[1]);

47                         sbItemName.append(" ").append(userIn[i]);

49                     String itemName = sbItemName.toString().toLowerCase();

51                         Item droppedItem = player.getInventory().getItem(itemName);

55                     }
